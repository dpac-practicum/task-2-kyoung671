import subprocess
import time
import logging

#Hit Ctrl + C in shell to stop the script from running and the log file will be generated afterwards

def ping():
    ping = subprocess.check_output(["ping","35.192.3.38","-c","1"])
    strPing = str(ping)
    logging.basicConfig(filename = 'ping.log',filemode = 'w',format='%(asctime)s - %(message)s', datefmt='%d-%b-%y %H:%M:%S')
    logger = logging.getLogger(__name__)
    logger.setLevel(logging.DEBUG)
    logger.info(strPing)
    #adjust number to however many seconds needed e.g. 900 for 30 minutes
    time.sleep(5)

def run(): 
    while True:
        try:
            ping()

        except subprocess.CalledProcessError as e:
            strPing = "Host Could not be reached"
            logging.basicConfig(filename = 'ping.log',filemode = 'w',format='%(asctime)s - %(message)s', datefmt='%d-%b-%y %H:%M:%S')
            logger = logging.getLogger(__name__)
            logger.setLevel(logging.DEBUG)
            logger.info(strPing)
            return "Go"

goAgain = run()

while goAgain == "Go":
    goAgain = run()
