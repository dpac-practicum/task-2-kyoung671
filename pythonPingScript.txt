import subprocess
import time
import logging

#Hit Ctrl + C in shell to stop the script from running
#If you are using Python IDLE you can check the log in real time
#If you are using Online GDB you have to stop the script before the log can be generated for viewing

def ping():
    ping = subprocess.check_output(["ping","35.192.3.38","-c","1"])
    strPing = str(ping)
    logging.basicConfig(filename='C:\\Users\\krs1_\\Documents\\ping.log',filemode = 'w',format='%(asctime)s - %(message)s', datefmt='%d-%b-%y %H:%M:%S')
    logger = logging.getLogger(__name__)
    logger.setLevel(logging.DEBUG)
    logger.info(strPing)
    #adjust number to however many seconds needed e.g. 900 for 30 minutes
    time.sleep(5)

def run(): 
    while True:
        try:
            ping()

        except subprocess.CalledProcessError as e:
            strPing = "Host Could not be reached"
            #for filename change to path where you'd like to create your logfile; use '\\' between directories e.g. 'c:\\users\\JohnDoe\\ping.log'
            logging.basicConfig(filename='PLACE PATH TO LOGFILE HERE',filemode= 'w',format='%(asctime)s - %(message)s', datefmt='%d-%b-%y %H:%M:%S')
            logger = logging.getLogger(__name__)
            logger.setLevel(logging.DEBUG)
            logger.info(strPing)
            #adjust number to however many seconds needed e.g. 900 for 30 minutes
            time.sleep(5)
            return "Go"

goAgain = run()

while goAgain == "Go":
    goAgain = run()
